The prime factors of 13195 are 5, 7, 13 and 29. What is the largest prime factor of the number 600851475143 ? 
var number = 600851475143,
    check = 2,
    array = [];
while (check * check <= number) {
    if (number % check == 0) {
        array.push(check);
        number = number / check;
    } else {
        check += 1;
    }
}
if (number != 1) {
    array.push(number);
}
// array is now full list of prime number factors

document.getElementById('answer').innerHTML = array[array.length - 1];
