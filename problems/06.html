Find the difference between the sum of the squares of the first one hundred natural numbers and the square of the sum. 
var result = 0,
    i,
    sumOfSquares = 0,
    squareOfSum = 0,
    squareRoot = function (n) {
        return n * n;
    };

for (i = 1; i <= 100; i += 1) {
    sumOfSquares += squareRoot(i);
    squareOfSum += i;
}
result = squareRoot(squareOfSum) - sumOfSquares;
