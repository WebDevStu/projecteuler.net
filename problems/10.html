The sum of the primes below 10 is 2 + 3 + 5 + 7 = 17. Find the sum of all the primes below two million. 
var result = 2,
    max = 2000000,
    i, sr, prime, p;
for (i = 3; i <= max; i += 2) {
    sr = Math.sqrt(i);
    prime = true;
    for (p = 3; prime && (p <= sr); p += 2) {
        if (i % p == 0) {
            prime = false;
        } else {
            prime = true;
        }
    }
    if (prime){
        result += i;
    }
}
