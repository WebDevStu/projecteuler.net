A Pythagorean triplet is a set of three natural numbers, a b c, for which, a2 + b2 = c2 For example, 32 + 42 = 9 + 16 = 25 = 52. There exists exactly one Pythagorean triplet for which a + b + c = 1000. Find the product abc. 
// the ultimate in brute force !!
var result = 0,
    a, b, c, 
    resa, resb, resc, 
    total = 500;
for (c = 1; c <= total; c += 1) {
    resc = Math.pow(c, 2);
    
    for (b = 1; b <= total; b += 1) {
        resb = Math.pow(b, 2);
        
        for (a = 1; a <= total; a += 1) {
            resa = Math.pow(a, 2);
            
            if ((a + b + c) === 1000 && (resa + resb) === resc) {
                result = (a * b * c)
                break;
            }
        }
    }
}
