A palindromic number reads the same both ways. The largest palindrome made from the product of two 2-digit numbers is 9009 = 91 99. Find the largest palindrome made from the product of two 3-digit numbers. 
var one = 999,
    two = 999,
    result = 0,
    o,
    t;
    reverseMatch = function (number) {
        return (number === parseFloat(('' + number).split('').reverse().join(''))) ? true : false;
    };                    
for (o = one; o > 100; o -= 1) {
    for (t = two; t > 100; t -= 1) {
        if (reverseMatch(o * t)) {
            ((o * t) > result) ? result = o * t : result = result;
        }
    }
}
